\section*{Timed automata}

\begin{exercise}
    \label{ex:1-TA}
    Show that the class of timed languages recognised by nondeterministic timed automata
    is not closed under complementation.
\end{exercise}
\begin{solution}
    Since the untiming of a regular timed language is regular,
    it suffices to find a regular timed language $L(A)$
    \st~the untiming of its complement $T(\Sigma) \setminus L(A)$ is not regular
    (this is a stronger result).
    %
    It suffices to consider a 1-clock nontederministic timed automaton $A$.
    Fix a two letter alphabet $\Sigma = \set{a, b}$.
    Consider the language $M$ of all words of the form $a^m b^n$, where $m \leq n$.
    A simple pumping argument shows that $M$ is not regular.

    We can construct a timed encoding of such words
    by requiring that every $a$ is follows by a $b$ after exactly one time unit.
    %
    A 1-clock nondeterministic timed automaton $A$ can verify words not satisfying this encoding.
    Amongst other cases, it can guess that an $a$ at time $t$ is not followed by a $b$ at time $t+1$.
\end{solution}

\begin{exercise}
    What is the complexity of the universality problem \emph{for the untiming}
    of timed languages recognised by nondeterministic timed automata?
\end{exercise}
\begin{solution}
    Clearly we obtain an exponential space upper bound
    by first untiming the automaton and then checking universality.
    The question is whether this is optimal.
    It turns out that the problem is indeed EXPSPACE-complete~\cite{BrenguierGollerSankur:CONCUR:2012}.
\end{solution}

\section*{Timed automata with $\e$-transitions}

\begin{exercise}
    Show that the class of timed languages recognised by
    timed automata with $\e$-transitions is strictly larger
    than the class of timed regular languages (\ie, without $\e$-transitions).
\end{exercise}
\begin{solution}
    We can encode the language $L$ of timed words over $\Sigma = \set a$
    of the form $\tuple{a, t}$ where $t$ is an integer.
    %
    A two states $Q = \set{p, q}$ 1-clock automaton with $\e$-transitions can recognise $L$.
    While waiting in the initial state $p$,
    the automaton resets its clock whenever $x = 1$ with an $\e$-transition.
    %
    Nondeterministically, it can also read $a$ when $x = 0$ and move to state $q$,
    which is accepting.

    Without $\e$-transitions, not even a timed Turing machine $A$ can recognise $L$.
    %
    By way of contradiction, suppose this is not the case
    and let $M$ be the maximal constant appearing in any transition of the machine.
    %
    Consider the input words $w = \tuple{a, M+1}$ and $w' = \tuple{a, M+1.5}$.
    Since $w \in L$, by assumption $A$ accepts $w$.
    After elapsing $M+1$ time units, the clocks of $A$ all have value $M+1$.
    Any clock guard of $A$ satisfied by elapsing $M+1$ time units
    is also satisfied by elapsing $M+1.5$ time units.
    Thus the same run of $A$ on $w$ also shows that $A$ accepts $w'$,
    which is a contradiction since $w' \notin L$.
\end{solution}

\begin{exercise}
    Are untimings of timed languages recognised by one-clock alternating timed automata regular?
\end{exercise}
\begin{solution}
    No, we can build a 1-clock timed automaton
    whose untiming is the (nonregular) language $M$ of all words of the form $a^m b^n$,
